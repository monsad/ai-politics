    ```python
    """FastAPI app: health + SSE stream endpoint for session transcripts."""
    from __future__ import annotations

    import json
    import os
    from pathlib import Path

    import aiosqlite
    from fastapi import FastAPI
    from sse_starlette.sse import EventSourceResponse

    app = FastAPI(title="Virtual Parliament API")


    def _db_path() -> Path:
        return Path(os.environ.get("PARLIAMENT_DB_PATH", "sessions.db"))


    @app.get("/health")
    def health() -> dict[str, str]:
        return {"status": "ok"}


    @app.get("/stream/{session_id}")
    async def stream_session(session_id: str):
        async def event_generator():
            db_path = _db_path()
            if not db_path.exists():
                yield {"event": "not_found", "data": json.dumps({"session_id": session_id, "reason": "no db"})}
                return
            async with aiosqlite.connect(str(db_path)) as db:
                async with db.execute(
                    "SELECT id FROM sessions WHERE id = ?", (session_id,)
                ) as cur:
                    row = await cur.fetchone()
                if row is None:
                    yield {"event": "not_found", "data": json.dumps({"session_id": session_id})}
                    return

                async with db.execute(
                    "SELECT seq, speaker, phase, content, node_ids "
                    "FROM utterances WHERE session_id = ? ORDER BY seq",
                    (session_id,),
                ) as cur:
                    async for row in cur:
                        yield {
                            "event": "utterance",
                            "data": json.dumps({
                                "seq": row[0],
                                "speaker": row[1],
                                "phase": row[2],
                                "content": row[3],
                                "node_ids": json.loads(row[4]) if row[4] else [],
                            }),
                        }

                async with db.execute(
                    "SELECT status, vote_result FROM sessions WHERE id = ?", (session_id,)
                ) as cur:
                    s = await cur.fetchone()
                    if s:
                        yield {"event": "status", "data": json.dumps({"status": s[0], "vote_result": s[1]})}

        return EventSourceResponse(event_generator())
    ```

    @pytest.mark.slow
    def test_full_simulation_4day_work_week(tmp_path, monkeypatch):
        """CLI-01 + ORCH-02..07 + political coherence."""
        from parliament.session import run_session
        from parliament.citation_validator import extract_node_ids, validate_citations
        import asyncio, time

        monkeypatch.setenv("PARLIAMENT_DB_PATH", str(tmp_path / "sessions.db"))
        start = time.monotonic()
        result = run_session("4-day work week", db_path=tmp_path / "sessions.db")
        elapsed = time.monotonic() - start

        assert result.returncode == 0, result.stderr[-2000:]
        assert elapsed <= 360, f"Session took {elapsed:.1f}s, exceeds 5min budget"
        assert "[MARSZAŁEK REASONING]" in result.stdout, "ORCH-08: reasoning blocks required"

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP client → /stream/{session_id} | Unauthenticated reads of session transcripts |
| HTTP path parameter → SQL | `session_id` flows to parameterised query |
| sessions.db → SSE response | Transcript content sent over the wire |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-05-01 | Tampering | SQL injection via session_id path param | mitigate | All SELECTs use `?` placeholders; `(session_id,)` tuple form |
| T-3-05-02 | Information Disclosure | SSE endpoint is unauthenticated | accept | Contest scope is local CLI + local Next.js demo on same machine; FastAPI binds 127.0.0.1 by default; auth is a Phase 5+ concern. Note documented in Phase 5 README. |
| T-3-05-03 | Spoofing | Session_id enumeration | accept | Session IDs are Hermes UUIDs (high entropy); enumeration impractical |
| T-3-05-04 | DoS | Large transcript blocks event loop | mitigate | aiosqlite is async; ORDER BY seq streams row-by-row, no full-table load |
| T-3-05-05 | DoS | Slow E2E tests cost real API spend | mitigate | Human checkpoint (Task 2) gates the spend; user can decline approval |
</threat_model>
